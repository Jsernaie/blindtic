                // Close mobile menu if open

                if (document.getElementById('nav-content').classList.contains('block')) {

        // Mobile menu toggle

            document.getElementById('nav-content').classList.toggle('hidden');

        };

 

        // Gemini AI Feature Logic

        const generateDescriptionBtn = document.getElementById('generateDescriptionBtn');

        const sceneDescriptionInput = document.getElementById('sceneDescription');

        const userQuestionInput = document.getElementById('userQuestion');

        const aiResponseDiv = document.getElementById('aiResponse');

        const loadingIndicator = document.getElementById('loadingIndicator');

        const errorMessageDiv = document.getElementById('errorMessage');

 

        generateDescriptionBtn.addEventListener('click', async () => {

            const scene = sceneDescriptionInput.value.trim();

            const question = userQuestionInput.value.trim();

 

            if (!scene && !question) {

                errorMessageDiv.textContent = 'Por favor, describe una escena o haz una pregunta.';

                errorMessageDiv.classList.remove('hidden');

                aiResponseDiv.classList.add('hidden');

                return;

            }

 

            errorMessageDiv.classList.add('hidden');

            aiResponseDiv.classList.add('hidden');

            loadingIndicator.classList.remove('hidden');

 

            try {

                let prompt = `Actúa como el asistente de IA del smartwatch BlindTic, diseñado para personas ciegas. Describe un entorno basado en la siguiente escena proporcionada por el usuario, y si se hace una pregunta específica, respóndela de manera clara y concisa como lo haría un asistente de voz. Prioriza la claridad y la información esencial.

 

Escena: "${scene}"

Pregunta/Solicitud: "${question}"

 

Tu respuesta debe ser una descripción auditiva.`;

 

                let chatHistory = [];

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""; // If you want to use models other than gemini-2.0-flash or imagen-3.0-generate-002, provide an API key here. Otherwise, leave this as-is.

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

 

                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

 

                const result = await response.json();

 

                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    aiResponseDiv.innerHTML = `<p class="font-bold mb-2">Respuesta de BlindTic:</p><p>${text}</p>`;

                    aiResponseDiv.classList.remove('hidden');

                } else {

                    errorMessageDiv.textContent = 'No se pudo generar una descripción. Por favor, inténtalo de nuevo.';

                    errorMessageDiv.classList.remove('hidden');

                    console.error('Unexpected API response structure:', result);

                }

 

            } catch (error) {

                errorMessageDiv.textContent = 'Ocurrió un error al comunicarse con la IA. Por favor, inténtalo más tarde.';

                errorMessageDiv.classList.remove('hidden');

                console.error('Error fetching from Gemini API:', error);

            } finally {

                loadingIndicator.classList.add('hidden');

            }

        });
